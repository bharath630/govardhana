document.getElementById('generateQrButton').onclick = function () { // Collect input values const adharNumber = document.getElementById('adhar').value; const donorName = document.getElementById('name').value; const donorPhone = document.getElementById('text').value; const donorEmail = document.getElementById('email').value; // Validate required inputs if (!adharNumber) { alert("Please enter your Aadhar number."); return; } if (!donorEmail) { alert("Please enter your mail id!"); return; } if (!donorName) { alert("Please enter your name!"); return; } if (!donorPhone) { alert("Please enter your phone number!"); return; } if (!formSubmitted) { alert("Please submit the form before generating the QR code."); return; } const amount = parseFloat(document.getElementById('amount').value); if (isNaN(amount) || amount <= 0) { alert("Invalid amount!"); return; } // Build UPI data string with URL encoding const upiData = `upi://pay?pa=govardhanylp@sbi&am=${encodeURIComponent(amount.toFixed(2))}&cu=INR&tn=${encodeURIComponent(donorPhone)}`; // Build QR code URL with encoded UPI data const qrCodeUrl = `https://api.qrserver.com/v1/create-qr-code/?size=400x400&data=${encodeURIComponent(upiData)}`; // Detect device and handle payment accordingly detectDeviceAndProceed(amount, adharNumber, qrCodeUrl, upiData); // Device detection and payment flow function function detectDeviceAndProceed(amount, adharNumber, qrCodeUrl, upiData) { const userAgent = navigator.userAgent.toLowerCase(); const isTouchScreen = 'ontouchstart' in window || navigator.maxTouchPoints > 0; const isAndroid = () => /android/.test(userAgent); const isIphone = () => /iphone|ipad|ipod/.test(userAgent) || (isTouchScreen && /mac os x/.test(userAgent) && navigator.maxTouchPoints > 1); const isDesktop = () => !isTouchScreen && (/windows/.test(userAgent) || /macintosh|mac os x/.test(userAgent)); if (isAndroid()) { // Redirect directly to UPI app (Android) window.location.href = upiData; } else if (isIphone()) { // Open payment options for iOS devices redirectToUPIAppOnIOS(amount); } else if (isDesktop()) { // Show QR code on desktop in modal or new window openQrWindow(qrCodeUrl); } else { // Default to QR code display openQrWindow(qrCodeUrl); } } // Payment options popup for iOS with encoded URLs function redirectToUPIAppOnIOS(amount) { const encodedAmount = encodeURIComponent(amount.toFixed(2)); const encodedPhone = encodeURIComponent(donorPhone); const appLinks = [ { name: "Google Pay", url: `tez://upi/pay?pa=govardhanylp@sbi&am=${encodedAmount}&tn=${encodedPhone}&cu=INR` }, { name: "PhonePe", url: `phonepe://pay?pa=govardhanylp@sbi&am=${encodedAmount}&tn=${encodedPhone}&cu=INR` }, { name: "Paytm", url: `paytmmp://upi/pay?pa=govardhanylp@sbi&am=${encodedAmount}&tn=${encodedPhone}&cu=INR` }, { name: "BHIM", url: `bhim://upi/pay?pa=govardhanylp@sbi&am=${encodedAmount}&tn=${encodedPhone}&cu=INR` }, ]; const optionsHtml = window.open('', '_blank', 'width=600,height=800'); optionsHtml.document.write(` Payment Options

Select preferred UPI App




${appLinks.map(app => ` ${app.name} `).join('')}
`); optionsHtml.document.close(); } // Show QR code in new window function openQrWindow(qrCodeUrl) { const qrWindow = window.open('', '_blank', 'width=500,height=600'); qrWindow.document.write(` UPI Payment QR Code

Scan to Pay

Scan this code to make the payment

`); qrWindow.document.close(); } };
